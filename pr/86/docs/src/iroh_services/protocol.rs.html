11/// The main ALPN for connecting from the client to the cloud node.
12///
13/// # Versioning policy
14///
15/// The wire protocol evolves **append-only** and does not bump the version in
16/// this ALPN for additive changes. postcard encodes enum variants by index, so
17/// as long as new [`IrohServicesProtocol`] request variants and new
18/// [`RemoteError`] variants are only ever *appended* (never inserted, reordered,
19/// or removed), every previously-defined message stays wire-compatible. That
20/// gives the following compatibility guarantees:
21///
22/// - **Old client → new server: fully compatible.** The server decodes every
23///   request an older client can send and only ever replies with error variants
24///   that client already knows.
25/// - **New client → old server:** the pre-existing requests (auth, metrics, …)
26///   still work; a *new* request the old server doesn't know fails as a
27///   per-request error rather than corrupting the connection or other traffic.
28///
29/// Because the cloud node is deployed at or ahead of the clients that talk to it,
30/// the second case is transient and confined to the new calls. A *breaking*
31/// change — reordering or removing variants, or changing a message's shape —
32/// would instead require a new ALPN.
33pub const ALPN: &[u8] = b"/iroh/n0des/1";

78pub type RemoteResult<T> = Result<T, RemoteError>;
79
80#[derive(Clone, Serialize, Deserialize, thiserror::Error, Debug)]
81#[non_exhaustive]
82pub enum RemoteError {
83    // The first three variants and their order are the v1 wire contract: postcard
84    // encodes enum variants by index, so a v1 client only decodes these and at
85    // their original positions. New variants MUST be appended after them, and the
86    // server must only send new variants in response to new (v2+) requests.
87    #[error("Missing capability: {}", _0.to_strings().join(", "))]
88    MissingCapability(Caps),
89    #[error("Unauthorized: {}", _0)]
90    AuthError(String),
91    #[error("Internal server error")]
92    InternalServerError,
93    #[error("Invalid input: {}", _0)]
94    InvalidInput(String),
95    #[error("Rate limit exceeded")]
96    RateLimited,
97}

161#[cfg(test)]
162mod tests {
163    use std::collections::BTreeMap;

165    use super::{RemoteError, SetAttributes, SetGroup};

168    #[test]
169    fn test_remote_error_wire_compat() {
170        // postcard encodes enum variants by their index. v1 clients only know
171        // the first three RemoteError variants, so these indices are a frozen
172        // wire contract; new variants must be appended after them.
173        let idx = |e: &RemoteError| postcard::to_stdvec(e).expect("encode")[0];
174        assert_eq!(idx(&RemoteError::AuthError(String::new())), 1);
175        assert_eq!(idx(&RemoteError::InternalServerError), 2);
176        // v2+ variants, appended after the v1 set.
177        assert_eq!(idx(&RemoteError::InvalidInput(String::new())), 3);
178        assert_eq!(idx(&RemoteError::RateLimited), 4);
179    }

181    // The wire format used by irpc (and elsewhere in this crate, see
182    // `api_secret.rs`) is postcard. These round-trips pin the on-the-wire
183    // contract these messages share with the server.
184
185    #[test]
186    fn test_set_group_serde_roundtrip() {
187        // a normal group, plus a unicode group for good measure
188        for group in ["staging", "my-group 👋"] {
189            let msg = SetGroup {
190                group: group.to_string(),
191            };
192            let bytes = postcard::to_stdvec(&msg).expect("postcard serialize");
193            let decoded: SetGroup = postcard::from_bytes(&bytes).expect("postcard deserialize");
194            assert_eq!(decoded.group, msg.group);
195        }
196    }
197
198    #[test]
199    fn test_set_attributes_serde_roundtrip() {
200        // empty map: the documented "clear" case
201        let empty = SetAttributes {
202            attributes: BTreeMap::new(),
203        };
204        let bytes = postcard::to_stdvec(&empty).expect("postcard serialize");
205        let decoded: SetAttributes = postcard::from_bytes(&bytes).expect("postcard deserialize");
206        assert!(decoded.attributes.is_empty());
207        assert_eq!(decoded.attributes, empty.attributes);
208
209        // unicode key/value plus a value at exactly the documented max length
210        let mut attributes = BTreeMap::new();
211        attributes.insert("région 🌍".to_string(), "us-wëst 🚀".to_string());
212        let max_value = "x".repeat(CLIENT_ATTRIBUTE_VALUE_MAX_LENGTH);
213        assert_eq!(max_value.len(), CLIENT_ATTRIBUTE_VALUE_MAX_LENGTH);
214        attributes.insert("max".to_string(), max_value);
215
216        let msg = SetAttributes { attributes };
217        let bytes = postcard::to_stdvec(&msg).expect("postcard serialize");
218        let decoded: SetAttributes = postcard::from_bytes(&bytes).expect("postcard deserialize");
219        assert_eq!(decoded.attributes, msg.attributes);
220    }
