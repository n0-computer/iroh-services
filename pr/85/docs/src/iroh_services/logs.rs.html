1//! Client-side log collection: a `tracing-subscriber` layer that writes
2//! structured log records to rolling files on the local filesystem, plus a
3//! reload handle that lets the cloud control the level filter at runtime.
4//!
5//! # The cloud is the source of truth for the level
6//!
7//! The level filter starts at `off`. No tracing events are captured until the
8//! cloud pushes a [`crate::protocol::SetLogLevel`] over the [`ClientHost`]
9//! channel. The cloud sends one immediately after authenticating a connected
10//! endpoint, derived from the per-endpoint `endpoint_log_settings` row (when
11//! present) plus the project default.
12//!
13//! [`ClientHost`]: crate::ClientHost
14//!
15//! # Files live on the device
16//!
17//! Records land in rolling JSON files under a caller-supplied directory.
18//! Operators view, ship, or aggregate them with whatever tooling they
19//! already use (`tail`, `journalctl`, `vector`, etc.).
20//!
21//! # Typical usage
22//!
23//! ```no_run
24//! use iroh_services::logs::{self, FileLoggerConfig};
25//!
26//! # fn main() -> anyhow::Result<()> {
27//! // Installs a global subscriber. The filter starts at `off`; once a
28//! // ClientHost connection is up, the cloud pushes a SetLogLevel that
29//! // raises it to whatever the dashboard says.
30//! let (collector, _guard) = logs::install(FileLoggerConfig::new("./logs"))?;
31//!
32//! // Hand the collector to ClientHost so it can call `collector.set_filter`
33//! // when the cloud pushes overrides.
34//! # let _ = collector;
35//! # Ok(())
36//! # }
37//! ```
38//!
39//! Compose with additional layers (for example, a stderr fmt layer) via
40//! [`layer`]:
41//!
42//! ```no_run
43//! use iroh_services::logs::{self, FileLoggerConfig};
44//! use tracing_subscriber::prelude::*;
45//!
46//! # fn main() -> anyhow::Result<()> {
47//! let (collector, file_layer, _guard) = logs::layer(FileLoggerConfig::new("./logs"))?;
48//! tracing_subscriber::registry()
49//!     .with(file_layer)
50//!     .with(tracing_subscriber::fmt::layer())
51//!     .try_init()
52//!     .ok();
53//! # let _ = collector;
54//! # Ok(())
55//! # }
56//! ```

73/// Errors that can occur while installing the log collector.
74#[derive(Debug, thiserror::Error)]
75pub enum InstallError {
76    /// The default tracing dispatcher is already set; install once at startup.
77    #[error("global tracing dispatcher is already set")]
78    AlreadyInstalled,
79    /// The supplied directives string was rejected by `EnvFilter`.
80    #[error("invalid filter directives: {0}")]
81    InvalidDirectives(String),
82    /// File logger setup failed (could not create directory, open appender,
83    /// etc.).
84    #[error("file logger setup failed: {0}")]
85    FileLogger(#[from] FileLoggerError),
86}

88/// Errors that can occur while changing the active filter at runtime.
89#[derive(Debug, thiserror::Error)]
90pub enum SetFilterError {
91    /// The supplied directives string was rejected by `EnvFilter`.
92    #[error("invalid filter directives: {0}")]
93    InvalidDirectives(String),
94    /// Reloading the filter failed because the subscriber went away.
95    #[error("reload handle is no longer valid")]
96    ReloadFailed,
97}

99/// Handle to the cloud-controlled tracing filter. Cheap to clone; all clones
100/// share the same backing reload handle.
101#[derive(Clone)]
102pub struct LogCollector {
103    inner: Arc<CollectorInner>,
104}
105
106struct CollectorInner {
107    reload_handle: reload::Handle<EnvFilter, Registry>,
108    revert_task: Mutex<Option<AbortOnDropHandle<()>>>,
109}
110
111/// Off-state directive. Nothing is captured until the cloud sends a
112/// `SetLogLevel` with something more permissive.
113const OFF_DIRECTIVES: &str = "off";
114
115impl LogCollector {
116    /// Sets the active filter directives. When `expires_in` is set,
117    /// schedules a revert after that duration. The revert target is
118    /// `revert_to` when supplied; `None` means revert to `off`.
119    pub fn set_filter(
120        &self,
121        directives: &str,
122        expires_in: Option<Duration>,
123        revert_to: Option<&str>,
124    ) -> Result<(), SetFilterError> {
125        let filter = EnvFilter::try_new(directives)
126            .map_err(|e| SetFilterError::InvalidDirectives(e.to_string()))?;
127        self.inner
128            .reload_handle
129            .reload(filter)
130            .map_err(|_| SetFilterError::ReloadFailed)?;
131
132        let mut guard = self.inner.revert_task.lock().expect("poisoned");
133        *guard = None;
134
135        if let Some(expires_in) = expires_in {
136            let collector = self.clone();
137            let revert_to = revert_to.map(str::to_string);
138            let handle: JoinHandle<()> = n0_future::task::spawn(async move {
139                n0_future::time::sleep(expires_in).await;
140                let target = revert_to.as_deref();
141                if let Err(err) = collector.revert(target) {
142                    warn!(?err, "failed to revert log filter");
143                }
144            });
145            *guard = Some(AbortOnDropHandle::new(handle));
146        }
147        Ok(())
148    }
149
150    /// Reverts the active filter to `to`, or to the off state when `to` is
151    /// `None`.
152    pub fn revert(&self, to: Option<&str>) -> Result<(), SetFilterError> {
153        let directives = to.unwrap_or(OFF_DIRECTIVES);
154        let filter = EnvFilter::try_new(directives)
155            .map_err(|e| SetFilterError::InvalidDirectives(e.to_string()))?;
156        self.inner
157            .reload_handle
158            .reload(filter)
159            .map_err(|_| SetFilterError::ReloadFailed)
160    }
161}

164    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
165        f.debug_struct("LogCollector").finish_non_exhaustive()
166    }

169/// Installs a global tracing subscriber whose only output is a rolling
170/// file appender under `config.dir`. The level filter starts at `off`; the
171/// cloud must push a `SetLogLevel` for any events to be captured.
172///
173/// Returns the [`LogCollector`] to hand to [`crate::ClientHost`] so it can
174/// apply runtime overrides, and a [`WorkerGuard`] the caller must hold for
175/// the lifetime of the process so the non-blocking writer flushes on exit.
176///
177/// Call exactly once at process start. For composition with other layers,
178/// use [`layer`].
179pub fn install(config: FileLoggerConfig) -> Result<(LogCollector, WorkerGuard), InstallError> {
180    let (collector, file_layer, guard) = layer(config)?;
181    tracing_subscriber::registry()
182        .with(file_layer)
183        .try_init()
184        .map_err(|_| InstallError::AlreadyInstalled)?;
185    debug!("iroh-services file logger installed");
186    Ok((collector, guard))
187}

189/// Builds the cloud-controlled file layer and its [`LogCollector`] without
190/// installing a global subscriber. Use this when composing the file layer
191/// with other layers; the returned layer is pre-wrapped in the reloadable
192/// filter so the cloud's `SetLogLevel` overrides take effect.
193pub fn layer(
194    config: FileLoggerConfig,
195) -> Result<
196    (
197        LogCollector,
198        impl Layer<Registry> + Send + Sync + 'static,
199        WorkerGuard,
200    ),
201    InstallError,
202> {
203    // `EnvFilter::try_new("off")` cannot fail; "off" is always valid.
204    let filter = EnvFilter::try_new(OFF_DIRECTIVES).expect("'off' is always a valid directive");
205    let (filter, reload_handle) = reload::Layer::new(filter);
206
207    let (file_layer, guard) = file_layer::<Registry>(config)?;
208    let layer = file_layer.with_filter(filter);
209
210    let inner = Arc::new(CollectorInner {
211        reload_handle,
212        revert_task: Mutex::new(None),
213    });
214    let collector = LogCollector { inner };
215    Ok((collector, layer, guard))
216}

230/// Errors raised when constructing the file logger.
231#[derive(Debug, thiserror::Error)]
232pub enum FileLoggerError {
233    /// Could not create the log directory or open the rolling appender.
234    #[error("file logger setup failed: {0}")]
235    Io(#[from] std::io::Error),
236    /// `tracing-appender`'s builder rejected the configuration (for example
237    /// an invalid filename prefix).
238    #[error("file logger builder rejected configuration: {0}")]
239    Builder(String),
240}

242/// Configuration for the rolling file logger.
243///
244/// Use [`FileLoggerConfig::new`] to set the destination directory and tune
245/// the remaining fields with the with-style setters. The defaults are
246/// daily rotation, a `iroh-services` filename prefix, and a 30-file
247/// retention window.
248#[derive(Debug, Clone)]
249pub struct FileLoggerConfig {
250    dir: PathBuf,
251    rotation: Rotation,
252    file_name_prefix: String,
253    max_files: Option<usize>,
254}
255
256impl FileLoggerConfig {
257    /// Build a config rooted at `dir`. The directory is created on first
258    /// write if it does not exist.
259    pub fn new<P: Into<PathBuf>>(dir: P) -> Self {
260        Self {
261            dir: dir.into(),
262            rotation: Rotation::DAILY,
263            file_name_prefix: "iroh-services".into(),
264            max_files: Some(30),
265        }
266    }
267
268    /// Override the rotation cadence. Default: [`Rotation::DAILY`].
269    pub fn with_rotation(mut self, rotation: Rotation) -> Self {
270        self.rotation = rotation;
271        self
272    }
273
274    /// Override the file name stem. Rotation appends a date suffix to this.
275    /// Default: `iroh-services`.
276    pub fn with_file_name_prefix<S: Into<String>>(mut self, prefix: S) -> Self {
277        self.file_name_prefix = prefix.into();
278        self
279    }
280
281    /// Override the retention cap. `None` keeps every file forever; `Some(n)`
282    /// keeps at most `n` files and deletes the oldest on rotation. Default:
283    /// `Some(30)`.
284    pub fn with_max_files(mut self, max_files: Option<usize>) -> Self {
285        self.max_files = max_files;
286        self
287    }
288}

290/// Builds an unfiltered tracing layer that writes records to a rolling
291/// file under `config.dir`. Returns the layer plus a [`WorkerGuard`] the
292/// caller must hold for the lifetime of the process — drop it at shutdown
293/// so any buffered records flush before exit.
294///
295/// Most callers want [`layer`] or [`install`] instead, which apply the
296/// cloud-controlled `EnvFilter` reload handle. Use this when you want a
297/// plain file appender with no cloud filter integration.
298pub fn file_layer<S>(
299    config: FileLoggerConfig,
300) -> Result<(impl Layer<S> + Send + Sync + 'static, WorkerGuard), FileLoggerError>
301where
302    S: Subscriber + for<'a> LookupSpan<'a>,
303{
304    let FileLoggerConfig {
305        dir,
306        rotation,
307        file_name_prefix,
308        max_files,
309    } = config;
310
311    create_dir_all(&dir)?;
312
313    let mut builder = tracing_appender::rolling::RollingFileAppender::builder()
314        .rotation(rotation)
315        .filename_prefix(file_name_prefix);
316    if let Some(max) = max_files {
317        builder = builder.max_log_files(max);
318    }
319    let appender = builder
320        .build(&dir)
321        .map_err(|e| FileLoggerError::Builder(e.to_string()))?;
322
323    let (writer, guard) = tracing_appender::non_blocking(appender);
324    let layer = tracing_subscriber::fmt::layer()
325        .with_writer(writer)
326        .with_ansi(false)
327        .json();
328    Ok((layer, guard))
329}
330
331fn create_dir_all(dir: &Path) -> Result<(), FileLoggerError> {
332    std::fs::create_dir_all(dir).map_err(FileLoggerError::Io)
333}

339    /// `file_layer` writes records to a file in the configured directory,
340    /// and the WorkerGuard flushes pending writes on drop.
341    #[test]
342    fn file_layer_writes_to_disk() {
343        use tracing::Dispatch;
344        use tracing_subscriber::{Registry, layer::SubscriberExt};
345
346        let tmp = tempfile::tempdir().unwrap();
347        let (layer, guard) = file_layer::<Registry>(
348            FileLoggerConfig::new(tmp.path())
349                .with_file_name_prefix("test")
350                .with_max_files(Some(2)),
351        )
352        .expect("file_layer setup");
353
354        let subscriber = Registry::default().with(layer);
355        let dispatch = Dispatch::new(subscriber);
356        tracing::dispatcher::with_default(&dispatch, || {
357            tracing::info!(target: "file_layer_test", "hello from the file logger");
358        });
359        drop(guard);
360
361        let mut found = false;
362        for entry in std::fs::read_dir(tmp.path()).unwrap() {
363            let entry = entry.unwrap();
364            if !entry.file_name().to_string_lossy().starts_with("test") {
365                continue;
366            }
367            let contents = std::fs::read_to_string(entry.path()).unwrap();
368            if contents.contains("hello from the file logger") {
369                found = true;
370                break;
371            }
372        }
373        assert!(found, "expected log line to be written to a test.* file");
374    }

376    /// The cloud-controlled `layer` starts captured-nothing and only writes
377    /// after `set_filter` raises the level. Verifies the reload handle is
378    /// wired to the file layer end-to-end.
379    #[tokio::test(flavor = "current_thread")]
380    async fn cloud_filter_controls_file_writes() {
381        use tracing::Dispatch;
382
383        let tmp = tempfile::tempdir().unwrap();
384        let (collector, log_layer, guard) =
385            layer(FileLoggerConfig::new(tmp.path()).with_file_name_prefix("controlled")).unwrap();
386
387        let subscriber = Registry::default().with(log_layer);
388        let dispatch = Dispatch::new(subscriber);
389        tracing::dispatcher::with_default(&dispatch, || {
390            // Captured nothing yet — filter is "off".
391            tracing::info!(target: "logtest", "before-set");
392
393            collector
394                .set_filter("info", None, None)
395                .expect("set_filter to info");
396            tracing::info!(target: "logtest", "after-set");
397        });
398        drop(guard);
399
400        let mut combined = String::new();
401        for entry in std::fs::read_dir(tmp.path()).unwrap() {
402            let entry = entry.unwrap();
403            if entry
404                .file_name()
405                .to_string_lossy()
406                .starts_with("controlled")
407            {
408                combined.push_str(&std::fs::read_to_string(entry.path()).unwrap());
409            }
410        }
411        assert!(
412            !combined.contains("before-set"),
413            "before-set should be filtered out, got: {combined}"
414        );
415        assert!(
416            combined.contains("after-set"),
417            "after-set should be written after set_filter, got: {combined}"
418        );
419    }
