31/// Client is the main handle for interacting with n0des. It communicates with
32/// n0des entirely through an iroh endpoint, and is configured through a builder.
33/// Client requires either an Ssh Key or [`ApiSecret`]
34///
35/// ```no_run
36/// use iroh_n0des::Client;
37///
38/// async fn build_client() -> anyhow::Result<()> {
39///     let endpoint = iroh::Endpoint::bind().await?;
40///
41///     // needs N0DES_API_SECRET set to an environment variable
42///     // client will now push endpoint metrics to n0des.
43///     let client = Client::builder(&endpoint)
44///         .api_secret_from_str("MY_API_SECRET")?
45///         .build()
46///         .await;
47///
48///     Ok(())
49/// }
50/// ```
51///
52/// [`ApiSecret`]: crate::api_secret::ApiSecret
53#[derive(Debug, Clone)]
54pub struct Client {
55    // owned clone of the endpoint for diagnostics, and for connection restarts on actor close
56    #[allow(dead_code)]
57    endpoint: Endpoint,
58    message_channel: tokio::sync::mpsc::Sender<ClientActorMessage>,
59    _actor_task: Arc<AbortOnDropHandle<()>>,
60}

120    /// set client API secret from an encoded string
121    pub fn api_secret_from_str(self, secret_key: &str) -> Result<Self> {

126    /// Use a shared secret & remote n0des endpoint ID contained within a ticket
127    /// to construct a n0des client. The resulting client will have "Client"
128    /// capabilities.
129    ///
130    /// API secrets include remote details within them, and will set both the
131    /// remote and rcan values on the builder
132    pub fn api_secret(mut self, ticket: ApiSecret) -> Result<Self> {
133        let local_id = self.endpoint.id();
134        let rcan = crate::caps::create_api_token_from_secret_key(
135            ticket.secret,
136            local_id,
137            self.cap_expiry,
138            Caps::for_shared_secret(),
139        )?;
140
141        self.remote = Some(ticket.remote);
142        self.rcan(rcan)
143    }

145    /// Loads the private ssh key from the given path, and creates the needed capability.
146    #[cfg(feature = "ssh-key")]
147    pub async fn ssh_key_from_file<P: AsRef<std::path::Path>>(self, path: P) -> Result<Self> {
148        let file_content = tokio::fs::read_to_string(path).await?;
149        let private_key = ssh_key::PrivateKey::from_openssh(&file_content)?;
150
151        self.ssh_key(&private_key)
152    }
153
154    /// Creates the capability from the provided private ssh key.
155    #[cfg(feature = "ssh-key")]
156    pub fn ssh_key(mut self, key: &ssh_key::PrivateKey) -> Result<Self> {
157        let local_id = self.endpoint.id();
158        let rcan = crate::caps::create_api_token_from_ssh_key(
159            key,
160            local_id,
161            self.cap_expiry,
162            Caps::all(),
163        )?;
164        self.cap.replace(rcan);
165
166        Ok(self)
167    }
168
169    /// Sets the rcan directly.
170    pub fn rcan(mut self, cap: Rcan<Caps>) -> Result<Self> {
171        ensure!(
172            EndpointId::from_verifying_key(*cap.audience()) == self.endpoint.id(),
173            "invalid audience"
174        );
175        self.cap.replace(cap);
176        Ok(self)
177    }

179    /// Sets the remote to dial, must be provided either directly by calling
180    /// this method, or through calling the api_secret builder methods.
181    pub fn remote(mut self, remote: impl Into<EndpointAddr>) -> Self {
182        self.remote = Some(remote.into());
183        self
184    }

186    /// Create a new client, connected to the provide service node
187    #[must_use = "dropping the client will silently cancel all client tasks"]
188    pub async fn build(self) -> Result<Client, BuildError> {
189        debug!("starting iroh-n0des client");
190        let remote = self.remote.ok_or(BuildError::MissingRemote)?;
191        let capabilities = self.cap.ok_or(BuildError::MissingCapability)?;
192
193        let conn = IrohLazyRemoteConnection::new(self.endpoint.clone(), remote, ALPN.to_vec());
194        let client = N0desClient::boxed(conn);
195
196        let (tx, rx) = tokio::sync::mpsc::channel(1);
197        let metrics_task = AbortOnDropHandle::new(n0_future::task::spawn(
198            ClientActor {
199                capabilities,
200                client,
201                session_id: Uuid::new_v4(),
202                authorized: false,
203                interval_metrics_enabled: self.metrics_interval.is_some(),
204            }
205            .run(self.registry, self.metrics_interval, rx),
206        ));
207
208        Ok(Client {
209            endpoint: self.endpoint,
210            message_channel: tx,
211            _actor_task: Arc::new(metrics_task),
212        })
213    }

216#[derive(thiserror::Error, Debug)]
217pub enum BuildError {
218    #[error("Missing remote endpoint to dial")]
219    MissingRemote,
220    #[error("Missing capability")]
221    MissingCapability,
222    #[error("Unauthorized")]
223    Unauthorized,
224    #[error("Remote error: {0}")]
225    Remote(#[from] RemoteError),
226    #[error("Rpc connection error: {0}")]
227    Rpc(irpc::Error),
228    #[error("Connection error: {0}")]
229    Connect(ConnectError),
230}
231
232impl From<irpc::Error> for BuildError {
233    fn from(value: irpc::Error) -> Self {
234        match value {
235            irpc::Error::Request {
236                source:
237                    irpc::RequestError::Connection {
238                        source: iroh::endpoint::ConnectionError::ApplicationClosed(frame),
239                        ..
240                    },
241                ..
242            } if frame.error_code == 401u32.into() => Self::Unauthorized,
243            value => Self::Rpc(value),
244        }
245    }

264    pub async fn ping(&self) -> Result<Pong, Error> {
265        let (tx, rx) = oneshot::channel();
266        self.message_channel

272            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))?
273            .map_err(Error::Remote)
274    }

291    /// Grant capabilities to a remote endpoint. Creates a signed RCAN token
292    /// and sends it to n0des for storage. The remote can then use this token
293    /// when dialing back to authorize its requests.
294    #[cfg(feature = "client_host")]
295    pub async fn grant_capability(
296        &self,
297        remote_id: EndpointId,
298        caps: impl IntoIterator<Item = impl Into<crate::caps::Cap>>,
299    ) -> Result<(), Error> {
300        let cap = crate::caps::create_grant_token(
301            self.endpoint.secret_key().clone(),
302            remote_id,
303            DEFAULT_CAP_EXPIRY,
304            Caps::new(caps),
305        )
306        .map_err(Error::Other)?;
307
308        let (tx, rx) = oneshot::channel();
309        self.message_channel
310            .send(ClientActorMessage::GrantCap {
311                cap: Box::new(cap),
312                done: tx,
313            })
314            .await
315            .map_err(|_| Error::Other(anyhow!("granting capability")))?;
316
317        rx.await
318            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))?
319    }

322    #[cfg(feature = "net_diagnostics")]
323    pub async fn net_diagnostics(&self, send: bool) -> Result<DiagnosticsReport, Error> {
324        let report = run_diagnostics(&self.endpoint).await?;
325        if send {

327            self.message_channel
328                .send(ClientActorMessage::PutNetworkDiagnostics {

334

337                .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))?;
338        }

343    /// Publish a [ticket] to n0des so others can find it by calling fetch_tickets.
344    /// Publishing uses n0des to act as a signaling server,e.
345    /// This API is intentionally designed around tickets to encourage using
346    /// more than one signaling mechanism, like QR codes, sharing action sheets,
347    /// copy-paste, etc. Publishing to n0des should give a smooth happy-path
348    /// experience when two nodes are online, while the ticket format will still
349    /// work if they can be shared by other means.
350    ///
351    /// Tickets are unique by {endpoint, name, type}. This means its possible to
352    /// have multiple tickets with the same name, but from different endpoints.
353    ///
354    /// A ticket will remain published as long as the endpoint that published
355    /// is online. Stale tickets be pruned within 4 metrics collection intervals
356    /// of an endpoint going offline.
357    ///
358    /// Tickets cannot be published if a metrics collection interval is disabled,
359    /// because n0des uses metrics publishing as a keepalive for ticket records
360    ///
361    /// [ticket]: https://docs.rs/iroh-tickets
362    #[cfg(feature = "tickets")]
363    pub async fn publish_ticket<T: Ticket>(&self, name: String, ticket: T) -> Result<(), Error> {
364        let ticket_kind = T::KIND.to_string();
365        let ticket_bytes = ticket.to_bytes();
366        let (tx, rx) = oneshot::channel();
367
368        self.message_channel
369            .send(ClientActorMessage::PublishTicket {
370                name,
371                ticket_kind,
372                ticket_bytes,
373                done: tx,
374            })
375            .await
376            .map_err(|_| Error::Other(anyhow!("publishing ticket")))?;
377
378        rx.await
379            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))?
380    }

403    /// Get a ticket from n0des by name.
404    #[cfg(feature = "tickets")]
405    pub async fn fetch_ticket<T: Ticket>(
406        &self,
407        name: String,
408    ) -> Result<Option<PublishedTicket<T>>, Error> {
409        let (tx, rx) = oneshot::channel();
410        self.message_channel
411            .send(ClientActorMessage::FetchTicket {
412                name,
413                ticket_kind: T::KIND.to_string(),
414                done: tx,
415            })
416            .await
417            .map_err(|_| Error::Other(anyhow!("fetching ticket")))?;
418
419        let res = rx
420            .await
421            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))??;
422
423        match res {
424            Some(td) => {
425                let ticket = T::from_bytes(&td.ticket_bytes).map_err(|e| {
426                    Error::Other(anyhow!("parsing n0des ticket get response: {:?}", e))
427                })?;
428
429                Ok(Some(PublishedTicket {
430                    name: td.name,
431                    ticket,
432                }))
433            }
434            None => Ok(None),
435        }
436    }

438    /// List tickets published to n0des.
439    #[cfg(feature = "tickets")]
440    pub async fn fetch_tickets<T: Ticket>(
441        &self,
442        offset: u32,
443        limit: u32,
444    ) -> Result<Vec<PublishedTicket<T>>, Error> {
445        let (tx, rx) = oneshot::channel();
446        self.message_channel
447            .send(ClientActorMessage::FetchTickets {
448                offset,
449                limit,
450                ticket_kind: T::KIND.to_string(),
451                done: tx,
452            })
453            .await
454            .map_err(|_| Error::Other(anyhow!("fetching tickets")))?;
455
456        let tickets = rx
457            .await
458            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))??
459            .iter()
460            .map(|td| {
461                let ticket = T::from_bytes(&td.ticket_bytes).map_err(|e| {
462                    Error::Other(anyhow!("parsing n0des tickets list response: {:?}", e))
463                })?;
464                Ok::<PublishedTicket<T>, Error>(PublishedTicket {
465                    name: td.name.clone(),
466                    ticket,
467                })
468            })
469            .collect::<Result<Vec<PublishedTicket<T>>, _>>()?;
470
471        Ok(tickets)
472    }

530struct ClientActor {

532    client: N0desClient,
533    session_id: Uuid,
534    authorized: bool,
535    #[allow(dead_code)]
536    interval_metrics_enabled: bool,

539impl ClientActor {
540    async fn run(
541        mut self,
542        registry: Registry,
543        interval: Option<Duration>,
544        mut inbox: tokio::sync::mpsc::Receiver<ClientActorMessage>,
545    ) {
546        let registry = Arc::new(RwLock::new(registry));
547        let mut encoder = Encoder::new(registry);
548        let mut metrics_timer = interval.map(|interval| n0_future::time::interval(interval));
549        trace!("starting client actor");
550        loop {
551            trace!("client actor tick");
552            tokio::select! {
553                biased;
554                Some(msg) = inbox.recv() => {
555                    match msg {
556                        ClientActorMessage::Ping{ done } => {
557                            let res = self.send_ping().await;
558                            if let Err(err) = done.send(res) {
559                                debug!("failed to send ping: {:#?}", err);
560                                self.authorized = false;
561                            }
562                        },
563                        ClientActorMessage::SendMetrics{ done } => {
564                            trace!("sending metrics manually triggered");
565                            let res = self.send_metrics(&mut encoder).await;
566                            if let Err(err) = done.send(res) {
567                                debug!("failed to push metrics: {:#?}", err);
568                                self.authorized = false;
569                            }
570                        }
571                        ClientActorMessage::GrantCap{ cap, done } => {
572                            let res = self.grant_cap(*cap).await;
573                            if let Err(err) = done.send(res) {
574                                warn!("failed to grant capability: {:#?}", err);
575                            }
576                        }
577                        #[cfg(feature = "net_diagnostics")]
578                        ClientActorMessage::PutNetworkDiagnostics{ report, done } => {
579                            let res = self.put_network_diagnostics(*report).await;
580                            if let Err(err) = done.send(res) {
581                                warn!("failed to publish network diagnostics: {:#?}", err);
582                            }
583                        }
584                        #[cfg(feature = "tickets")]
585                        ClientActorMessage::PublishTicket{ name, ticket_kind, ticket_bytes, done } => {
586                            let res = self.tickets_publish(name, ticket_kind, ticket_bytes).await;
587                            if let Err(err) = done.send(res) {
588                                warn!("failed to publish ticket: {:#?}", err);
589                            }
590                        }
591                        #[cfg(feature = "tickets")]
592                        ClientActorMessage::UnpublishTicket{ name, ticket_kind, done } => {
593                            let res = self.tickets_unpublish(name, ticket_kind).await;
594                            if let Err(err) = done.send(res) {
595                                warn!("failed to unpublish ticket: {:#?}", err);
596                            }
597                        }
598                        #[cfg(feature = "tickets")]
599                        ClientActorMessage::FetchTicket{ name, ticket_kind, done } => {
600                            let res = self.ticket_fetch(name, ticket_kind).await;
601                            if let Err(err) = done.send(res) {
602                                warn!("failed to fetch tickets: {:#?}", err);
603                            }
604                        }
605                        #[cfg(feature = "tickets")]
606                        ClientActorMessage::FetchTickets{ done, ticket_kind, offset, limit } => {
607                            let res = self.tickets_fetch(ticket_kind, offset, limit).await;
608                            if let Err(err) = done.send(res) {
609                                warn!("failed to fetch tickets: {:#?}", err);
610                            }
611                        }
612                    }
613                }
614                _ = async {
615                    if let Some(ref mut timer) = metrics_timer {
616                        timer.tick().await;
617                    } else {
618                        std::future::pending::<()>().await;
619                    }
620                } => {
621                    trace!("metrics send tick");
622                    if let Err(err) = self.send_metrics(&mut encoder).await {
623                        debug!("failed to push metrics: {:#?}", err);
624                        self.authorized = false;
625                    }
626                },
627            }
628        }
629    }

631    // sends an authorization request to the server
632    async fn auth(&mut self) -> Result<(), RemoteError> {
633        if self.authorized {
634            return Ok(());
635        }
636        trace!("client authorizing");
637        self.client
638            .rpc(Auth {
639                caps: self.capabilities.clone(),
640            })
641            .await
642            .inspect_err(|e| debug!("authorization failed: {:?}", e))
643            .map_err(|e| RemoteError::AuthError(e.to_string()))?;
644        self.authorized = true;
645        Ok(())
646    }
647
648    async fn send_ping(&mut self) -> Result<Pong, RemoteError> {
649        trace!("client actor send ping");
650        self.auth().await?;
651
652        let req = rand::random();
653        self.client
654            .rpc(Ping { req_id: req })
655            .await
656            .inspect_err(|e| warn!("rpc ping error: {e}"))
657            .map_err(|_| RemoteError::InternalServerError)
658    }
659
660    async fn send_metrics(&mut self, encoder: &mut Encoder) -> Result<(), RemoteError> {
661        trace!("client actor send metrics");
662        self.auth().await?;
663
664        let update = encoder.export();
665        // let delta = update_delta(&self.latest_ackd_update, &update);
666        let req = PutMetrics {
667            session_id: self.session_id,
668            update,
669        };
670
671        self.client
672            .rpc(req)
673            .await
674            .map_err(|_| RemoteError::InternalServerError)??;
675
676        Ok(())
677    }

814    #[tokio::test]
815    #[temp_env_vars]
816    async fn test_api_key_from_env() {
817        // construct
818        let mut rng = rand::rng();
819        let shared_secret = SecretKey::generate(&mut rng);
820        let fake_endpoint_id = SecretKey::generate(&mut rng).public();
821        let api_secret = ApiSecret::new(shared_secret.clone(), fake_endpoint_id);
822        unsafe {
823            std::env::set_var(API_SECRET_ENV_VAR_NAME, api_secret.to_string());
824        };
825
826        let endpoint = Endpoint::empty_builder(iroh::RelayMode::Disabled)
827            .bind()
828            .await
829            .unwrap();
830
831        let builder = Client::builder(&endpoint).api_secret_from_env().unwrap();
832
833        let fake_endpoint_addr: EndpointAddr = fake_endpoint_id.into();
834        assert_eq!(builder.remote, Some(fake_endpoint_addr));
835
836        // Compare capability fields individually to avoid flaky timestamp
837        // mismatches between the builder's rcan and a freshly-created one.
838        let cap = builder.cap.as_ref().expect("expected capability to be set");
839        assert_eq!(cap.capability(), &Caps::new([Cap::Client]));
840        assert_eq!(cap.audience(), &endpoint.id().as_verifying_key());
841        assert_eq!(cap.issuer(), &shared_secret.public().as_verifying_key());
842    }
