1//! Minimal parser for unencrypted OpenSSH ed25519 private keys.
2//!
3//! See <https://github.com/openssh/openssh-portable/blob/master/PROTOCOL.key>
4//! for the on-disk format.
5
6use anyhow::{Context, Result, bail, ensure};
7use base64::{Engine as _, engine::general_purpose::STANDARD};
8
9const MAGIC: &[u8] = b"openssh-key-v1\0";
10const PEM_BEGIN: &str = "-----BEGIN OPENSSH PRIVATE KEY-----";
11const PEM_END: &str = "-----END OPENSSH PRIVATE KEY-----";
12
13/// Parse an unencrypted OpenSSH ed25519 private key (PEM-encoded) and return
14/// the 32-byte ed25519 seed.
15pub(crate) fn parse_ed25519_private_key(pem: &str) -> Result<[u8; 32]> {
16    let begin = pem.find(PEM_BEGIN).context("missing OpenSSH PEM header")?;
17    let after_header = begin + PEM_BEGIN.len();
18    let end_offset = pem[after_header..]
19        .find(PEM_END)
20        .context("missing OpenSSH PEM footer")?;
21    let body: String = pem[after_header..after_header + end_offset]
22        .chars()
23        .filter(|c| !c.is_whitespace())
24        .collect();
25    let bytes = STANDARD
26        .decode(body.as_bytes())
27        .context("invalid base64 in OpenSSH key")?;
28
29    let mut r = Reader::new(&bytes);
30    ensure!(r.take(MAGIC.len())? == MAGIC, "not an OpenSSH v1 key");
31    let cipher = r.string()?;
32    ensure!(
33        cipher == b"none",
34        "encrypted OpenSSH keys are not supported"
35    );
36    let kdf = r.string()?;
37    ensure!(kdf == b"none", "OpenSSH key has unexpected kdf");
38    let _kdf_options = r.string()?;
39    let nkeys = r.u32()?;
40    ensure!(nkeys == 1, "expected exactly one OpenSSH key, got {nkeys}");
41    let _public_key = r.string()?;
42    let private_section = r.string()?;
43
44    let mut r = Reader::new(private_section);
45    let c1 = r.u32()?;
46    let c2 = r.u32()?;
47    ensure!(c1 == c2, "OpenSSH checkint mismatch (key may be encrypted)");
48    let keytype = r.string()?;
49    ensure!(
50        keytype == b"ssh-ed25519",
51        "only ed25519 OpenSSH keys are supported"
52    );
53    let _public = r.string()?;
54    let private = r.string()?;
55    ensure!(
56        private.len() == 64,
57        "unexpected ed25519 private key length: {}",
58        private.len()
59    );
60
61    let mut seed = [0u8; 32];
62    seed.copy_from_slice(&private[..32]);
63    Ok(seed)
64}
65
66struct Reader<'a> {
67    buf: &'a [u8],
68}
69
70impl<'a> Reader<'a> {
71    fn new(buf: &'a [u8]) -> Self {
72        Self { buf }
73    }
74
75    fn take(&mut self, n: usize) -> Result<&'a [u8]> {
76        if self.buf.len() < n {
77            bail!("truncated OpenSSH key");
78        }
79        let (head, tail) = self.buf.split_at(n);
80        self.buf = tail;
81        Ok(head)
82    }
83
84    fn u32(&mut self) -> Result<u32> {
85        let b = self.take(4)?;
86        Ok(u32::from_be_bytes([b[0], b[1], b[2], b[3]]))
87    }
88
89    fn string(&mut self) -> Result<&'a [u8]> {
90        let len = self.u32()? as usize;
91        self.take(len)
92    }
93}
